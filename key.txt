name: Build and deploy to Azure Web App - cloudsprint3-rm556620

on:
  push:
    branches: ["main"]
  workflow_dispatch:

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Set up JDK 17 (Temurin) with Maven cache
        uses: actions/setup-java@v4
        with:
          java-version: '17'
          distribution: 'temurin'
          cache: 'maven'

      # Gera o fat jar (Spring Boot) â€” tests pulados p/ acelerar
      - name: Build with Maven (skip tests)
        run: mvn -B -DskipTests package

      # Escolhe o fat jar (evita *original*.jar)
      - name: Find Spring Boot fat JAR
        id: findjar
        run: |
          ls -la target
          BOOT_JAR=$(ls target/*.jar | grep -v original | head -n 1)
          echo "bootjar=$BOOT_JAR" >> $GITHUB_OUTPUT

      - name: Deploy to Azure Web App
        uses: azure/webapps-deploy@v2
        with:
          app-name: 'cloudsprint3-rm556620'
          publish-profile: ${{ secrets.AZURE_WEBAPP_PUBLISH_PROFILE }}
          package: '${{ steps.findjar.outputs.bootjar }}'
